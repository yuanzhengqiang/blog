最长递增子序列

# 动态规划

var lengthOfLIS = function(nums) {
    const temp = []
    let maxCount = 1
    for(let i = 0; i < nums.length; i++) {
        temp[i] = 1
        for(let j = 0; j < i; j++) {
            if (nums[i] > nums[j]) {
                temp[i] = Math.max(temp[i], temp[j]+1) 
            }
        }
        maxCount = Math.max(maxCount, temp[i])
    }
    return maxCount
};

# 二分查找

var lengthOfLIS = function(nums) {
    let n = nums.length;
    if(n <= 1){
        return n;
    }
    let tail = [nums[0]];
    for(let i = 0;i < n;i++){
        if(nums[i] > tail[tail.length-1]){
            tail.push(nums[i]);
        }else{
            let left = 0;
            let right = tail.length-1;
            while(left < right){
                let mid = (left + right) >> 1;
                if(tail[mid] < nums[i]){
                    left = mid + 1;
                }else{
                    right = mid;
                }
            }
            tail[left] = nums[i];
        }
    }
    return tail.length;
};
